        function removeDuplicate(str,n){
            var s = new Set();
            for(let i = 0; i < n; i++){
                s.add(str[i]);
            }
            for(const element of s){
                console.log(element);
            }
        }
        var str = "abcadeecfb"
        var n = str.length;
        removeDuplicate(str,n);
