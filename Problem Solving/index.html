        var Check_Leap = (year) => 
        {

            return "Leap Year";
          }
          else{
            return "Non Leap Year";

3. Perfect Number Check.

        var Perfect_Check = (N) => 
        {
          var sum = 0;
             for(let i = 0; i < N-1; i++){
               if(N % i ===0){
                 sum = sum + i;
               }
             } 
             if(sum === N){
               return "YES";
             }
             else{
               return "NO";
             }
                
        };
