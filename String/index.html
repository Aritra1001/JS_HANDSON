2. Count the head

        var Count_Occ = (s) => 
        {
            let n = s.length;
            let str = s.split("").sort().join("");
            let ans="";
            for(let i = 0; i < n; i++){
              let count = 0;
              for(let  j = 0; j < n; j++){
                if(str[i]===str[j] && i>j){
                  break;
                }
                if(str[i]===str[j]){
                  count++;
                }
              }
              if(count>1){
                ans+=str[i]+count;
              }
            }
            return ans;
        };

        var Palin_Check = (str) =>
        {
             var revStr ="";

               revStr = revStr + str[i];
             }
             if(revStr===str){

             }

             }
        }
