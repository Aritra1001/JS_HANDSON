4. EvenOdd

        const findEvenOdd = (N, Arr) => 
        {
             var sum_even = 0, sum_odd = 0;
             var B = [];
             let i = 0;
             for(i = 0; i < N ; i++)
             {
                 if(Arr[i] % 2 === 0)
                 {
                     sum_even = sum_even + Arr[i];
                 }
                 else
                 {
                     sum_odd = sum_odd + Arr[i];
                 }
             }
             for(i = 0; i < 2; i++)
             {
                 B = [sum_even, sum_odd];
             }
             return B;
        };

10. Min and Max

        function arrayMin(arr) {
            let min = arr[0];
            for(let i = 0; i < arr.length; i++){
              if(min > arr[i+1]){
                min = arr[i+1];
              }
            }
            return min;
          }
          
          function arrayMax(arr) {
            let max = arr[0];
            for(let j = 0; j < arr.length; j++){
              if(max < arr[j+1]){
                max = arr[j+1];
              }
            }
            return max;
          }

11. Birthday Game

        function Birthday_Game(arr,D ,M) {
            let count = 0;
            for(let i = 0; i < arr.length-M+1; i++)
            {
              let sum=0;
              for(let j=i; j < M+i; j++){
                sum+=arr[j];
              }
                if(sum === D)
                {
                  count++;
                }
              
            }
           return count;
